* 데이터 분석할 때 data의 개수를 모를때 range(len()) 사용
import random
n = 0
a = [1,2,3,4,5,6,7]
a1 = [1,2,3,4,5,6,7]
random.shuffle(a1)

for i in range(len(a)):
    if a[i] == a1[i]:
        n += 1
    else:
        pass

print(n)

-> 개발자는 range(len())대신 enumerate(a)를 자주 사용
import random
n = 0
a = [1,2,3,4,5,6,7]
b = [1,2,3,4,5,6,7]
random.shuffle(b)

for w, i in enumerate(a):      #i는 a 순서대로 입력, w에는 index 번호가 입력
    if a[w]== b[w]:              #a[w]는 i와 같기 때문에 i를 입력해도 됨
        n += 1
    else:
        pass

print(n)
